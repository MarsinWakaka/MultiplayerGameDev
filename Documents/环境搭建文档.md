* 序列化方案：MessagePack 
    * C# 项目直接通过Nuget包安装即可，Unity中需要先通过PackageManager安装Nuget for Unity 4.5.0,然后在Nuget工具中搜索安装MessagePack。
    * Nuget地址：https://github.com/GlitchEnzo/NuGetForUnity.git?path=/src/NuGetForUnity
    * 使用案例：[MessagePack For C#在Unity中的应用（场景的保存和加载） - 代码先锋网](https://www.codeleading.com/article/1578960766/)

```C#
// 添加第三方类型序列化        
var resolver = MessagePack.Resolvers.CompositeResolver.Create(
            new IMessagePackFormatter[]
            {
                new UnityVector3Formatter(),
                new UnityQuaternionFormatter()
            },
            new IFormatterResolver[]
            {
                MessagePack.Resolvers.StandardResolver.Instance
            }
        );
// 保存至默认方案
MessagePackSerializer.DefaultOptions = MessagePackSerializerOptions.Standard.WithResolver(resolver);
```

```C#
using MessagePack;
using MessagePack.Formatters;
using UnityEngine;

namespace MessagePackExtension
{
    public class UnityQuaternionFormatter : IMessagePackFormatter<Quaternion>
    {
        public void Serialize(ref MessagePackWriter writer, Quaternion value, MessagePackSerializerOptions options)
        {
            // writer.WriteArrayHeader(4);
            writer.Write(value.x);
            writer.Write(value.y);
            writer.Write(value.z);
            writer.Write(value.w);
        }

        public Quaternion Deserialize(ref MessagePackReader reader, MessagePackSerializerOptions options)
        {
            // var length = reader.ReadArrayHeader();
            var x = reader.ReadSingle();
            var y = reader.ReadSingle();
            var z = reader.ReadSingle();
            var w = reader.ReadSingle();
            return new Quaternion(x, y, z, w);
        }
    }
```
